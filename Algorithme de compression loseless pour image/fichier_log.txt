def reecriture(positions: list)->str:
    elements = list(set(positions)) #uniques elements
    phrase_fractionnee = ["" for _ in range(len(positions))] #remplie psk le remplissage ne se fera pas de proche en proche: position 24 peut être occupée avant position 7
    phrase_fractionnee[0]=str(positions[0])
    for element in elements[1:]: #Le premier element est tjrs seul!!
        index_de_debut_de_serie=[]
        for i in range(len(positions)):
            if positions[i] == element:
                if i == 0 or positions[i-1] != element:
                   index_de_debut_de_serie.append(i)
        for i in index_de_debut_de_serie:
            compteur = 1
            j = i + 1
            while j < len(positions) and positions[j] == element:
                compteur += 1
                j += 1
            if compteur > 1:
                phrase_fractionnee[i] = f"{element}*{compteur}" 
            else: phrase_fractionnee[i]=f"{element}"
    phrase_fractionnee = [nombre for i, nombre in enumerate(phrase_fractionnee) if nombre]
    phrase = ''.join(phrase_fractionnee)
    return phrase
